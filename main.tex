\newcommand{\myListLayoutEDU}[4]{ 
% #1: Institution, district and country  | #2: Date | #3: Course | #4: GPA
\noindent
\newline
\begin{minipage}{0.8\textwidth}
    \begin{flushleft}
        #1
    \end{flushleft}
\end{minipage}%
\begin{minipage}{0.2\textwidth}
    \begin{flushright}
        #2
    \end{flushright}
\end{minipage}

{\centering
\begin{minipage}{0.8\textwidth}
    \begin{flushleft}
        #3
    \end{flushleft}
\end{minipage}%
\begin{minipage}{0.2\textwidth}
    \begin{flushright}
        #4
    \end{flushright}
\end{minipage}}
}

\newcommand{\myListLayoutEXP}[3]{ 
% #1: Task | #2: Date | #3: Institution
\noindent
\newline
\begin{minipage}{0.8\textwidth}
    \begin{flushleft}
        #1
    \end{flushleft}
\end{minipage}%
\begin{minipage}{0.2\textwidth}
    \begin{flushright}
        #2
    \end{flushright}
\end{minipage}

{\raggedleft
    #3
}
}

\newcommand{\myListLayoutPROJ}[2]{ 
% #1: Project name & Date of birth | #2: Description
    \noindent
    \textbf{#1}\\
        #2\\
    \noindent
    \newline
}

\newcommand{\myListLayoutOneLIST}[2]{ 
% #1: Subject | #2: List
    \noindent
    \textbf{#1}
        \begin{itemize}
            \item #2
        \end{itemize}
}

\newcommand{\myListLayoutTwoLIST}[3]{ 
% #1: Subject | #2: List 1 | #3: List 2
    \noindent
    \textbf{#1}
        \begin{itemize}
            \item #2
            \item #3
        \end{itemize}
}
